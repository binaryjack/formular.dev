// 1. Import the library and create service manager
import { 
    ServiceManagerFactory, FormularManagerBuilder, 
    Validators, IFieldDescriptor 
} from 'formular.dev.lib'

// Create service manager using the new factory system
const serviceManager = ServiceManagerFactory.createDefault()
const formularManager = serviceManager.getService('IFormularManager')

// Create validation rules using the builder pattern
const emailValidation = new FormularManagerBuilder()
    .addValidation(Validators.required('Email is required'))
    .addValidation(Validators.email('Please enter a valid email'))
    .build()

const phoneValidation = new FormularManagerBuilder()
    .addValidation(Validators.phoneNumber(['CH', 'US', 'FR']))
    .build()

// Create field descriptors using the updated API
const emailField: IFieldDescriptor = {
    name: 'email',
    label: 'Email Address',
    type: 'email',
    validation: emailValidation,
    required: true
}

const phoneField: IFieldDescriptor = {
    name: 'phone',
    label: 'Phone Number',
    type: 'tel',
    validation: phoneValidation,
    required: false
}

// Create field descriptor for date picker
const dateField: IFieldDescriptor = {
    name: 'date',
    label: 'Date',
    type: 'date',
    validation: new FormularManagerBuilder()
        .addValidation(Validators.required('Date is required'))
        .build(),
    required: true
}

// Add date picker field to form descriptors
const formular = formularManager.createFromDescriptors(
    'user-registration', 
    [emailField, phoneField, dateField]
)

// Bind to existing HTML form with modern API
const formElement = document.getElementById('user-registration-form')
formular.bindToDOM(formElement)

⚛️ React Implementation (Available Now) 
// React adapter with modern service-based architecture (in development)
import { useService } from '@adapters/react/services/use-service'
import { useField } from '@adapters/react/fields/hooks/use-field'
import FormularForm from '@components/formular-form/formular-form'
import InputText from '@components/input-text/input-text'
import { ServiceManagerFactory } from 'formular.dev.lib'

function UserRegistrationForm() {
    const { getService } = useService()
    const formularManager = getService('IFormularManager')

    // Create form instance using IoC container
    const formular = formularManager?.createFromDescriptors(
        'user-registration', 
        [emailField, phoneField]
    )

    const handleSubmit = (data) => {
        console.log('Form submitted:', data)
        // Form data is automatically validated and typed
    }

    return (
        <FormularForm formular={formular} onSubmit={handleSubmit}>
            <InputText fieldName="email" />
            <InputText fieldName="phone" />
            <label for="date">Date</label>
            <input type="date" id="date" name="date" required />
            <div id="date-errors" class="error-container"></div>
            <button type="submit" disabled={!formular?.isValid}>
                Submit
            </button>
        </FormularForm>
    )
}

// Pure JavaScript integration with service container
import { ServiceManagerFactory, FormularManagerBuilder } from 'formular.dev.lib'

// Create service container with dependency injection
const serviceManager = ServiceManagerFactory.createDefault()
const formularManager = serviceManager.getService('IFormularManager')
const notificationManager = serviceManager.getService('INotificationManager')

// Create form instance from descriptors
const formular = formularManager.createFromDescriptors(
    'user-registration', 
    [emailField, phoneField]
)

// Bind to existing HTML form with modern API
const formElement = document.getElementById('user-registration-form')
formular.bindToDOM(formElement)

// Listen to form events using reactive notifications
notificationManager.subscribe('field:validation', (event) => {
    const { field, isValid, errors } = event.data
    const errorContainer = document.querySelector(`#${field.name}-errors`)
    
    if (errorContainer) {
        errorContainer.innerHTML = errors.map(err => 
            `<span class="error-message">${err.message}</span>`
        ).join('')
    }
})

// Handle form submission with automatic validation
notificationManager.subscribe('form:submit', (event) => {
    if (event.data.isValid) {
        console.log('Form submitted:', event.data.formData)
        // Process valid form data
    }
})
